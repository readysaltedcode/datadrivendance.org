              var width = 1000, height = 600;

              var yScale = d3.scale.linear().domain([-1.5, 1.5]).range([height, 0]);

              var ctx = canvas.getContext('2d');

              ctx.fillStyle = "rgb(0,0,0)";

                      var y = yScale(vertex.Y);
                      ctx.lineTo(x, y);
                    });
                  });
             
                  // Draw head

                  });

              }
